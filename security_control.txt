import requests

def check_remote_security_status(url="https://raw.githubusercontent.com/plantaloky/RemotePilot/main/security_control.txt"):
    try:
        response = requests.get(url)
        if response.status_code == 200:
            status = response.text.strip().upper()
            return status == "ON"
        else:
            print("Failed to fetch status. HTTP", response.status_code)
            return False
    except Exception as e:
        print("Error checking remote status:", e)
        return False

if check_remote_security_status():
    print("✅ Seguridad ACTIVADA (ON).")
else:
    print("⚠️ Seguridad DESACTIVADA (OFF).")
